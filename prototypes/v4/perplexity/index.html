#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
Build city_data_export.json for the web app.
Inputs (place alongside this script):
  - city_meta.json           # list of city objects with id, name, county, population, area and social media fields
  - transposed_scores.json   # preferred (normalized 0-100). Either {"cities": [{"id":..., "scores": {...}}]} or {id: {cat: score}}
  - city_scores.json         # fallback (normalized 0-100) with same shape as above
Output:
  - city_data_export.json    # {"cities": [...]} as required by the app
"""

import json
from pathlib import Path

ALL_CATEGORIES = [
    'ATMs','Books','Bus stations','Churches','Cinema','Clubs (dancing)',
    'Dental','Dentist','Dentist + Clinic','Doctor','Fast Food','Gyms Fitness',
    'Hospitals','Hotels','Lodging','Museums','Other Places of Worship',
    'Pharmacies','Pub Cafe','Restaurants','Schools','Shopping','University',
    'beauty wellness','culture_centers','dental_clinic','event_venue','gov',
    'post_office','shopping_mall','sports','swimming_pool','theatre','veterinary_animals'
]

HERE = Path(__file__).parent

def load_json(p: Path):
    if p.exists():
        with p.open('r', encoding='utf-8') as f:
            return json.load(f)
    return None

# Normalize scores container into {id: {category: score}}
def normalize_scores(raw):
    if raw is None:
        return {}
    # Case 1: {"cities": [{"id":, "scores": {...}}]}
    if isinstance(raw, dict) and 'cities' in raw and isinstance(raw['cities'], list):
        out = {}
        for item in raw['cities']:
            cid = item.get('id')
            scores = item.get('scores', {}) or {}
            if cid is not None:
                out[str(cid)] = scores
        return out
    # Case 2: mapping {id: {cat: score}}
    if isinstance(raw, dict):
        # Ensure keys are strings
        return {str(k): v for k, v in raw.items() if isinstance(v, dict)}
    # Case 3: list of {id: ..., scores: {...}}
    if isinstance(raw, list):
        out = {}
        for item in raw:
            if not isinstance(item, dict):
                continue
            cid = item.get('id')
            scores = item.get('scores', {}) or {}
            if cid is not None:
                out[str(cid)] = scores
        return out
    return {}

city_meta = load_json(HERE / 'city_meta.json') or []
transposed_scores = normalize_scores(load_json(HERE / 'transposed_scores.json'))
city_scores = normalize_scores(load_json(HERE / 'city_scores.json'))

scores_by_id = transposed_scores if transposed_scores else city_scores

cities_out = []
for m in city_meta:
    cid = str(m.get('id', m.get('city_id', '')))
    meta_social = {
        'official website': m.get('official website') or m.get('official_website') or '',
        'YouTube handle': m.get('YouTube handle') or m.get('youtube_handle') or '',
        'YouTube channel ID': m.get('YouTube channel ID') or m.get('youtube_channel_id') or '',
        'Facebook location ID': m.get('Facebook location ID') or m.get('facebook_location_id') or '',
        'Facebook username': m.get('Facebook username') or m.get('facebook_username') or ''
    }

    score_map = {cat: 0 for cat in ALL_CATEGORIES}
    if cid in scores_by_id:
        for cat, val in scores_by_id[cid].items():
            try:
                v = float(val)
            except Exception:
                continue
            # Clamp to 0..100
            v = max(0.0, min(100.0, v))
            if cat in score_map:
                score_map[cat] = v
            else:
                # If category label differs slightly, keep it as-is too
                score_map[cat] = v

    cities_out.append({
        'id': m.get('id'),
        'name': m.get('name') or m.get('city') or '',
        'county': m.get('county') or '',
        'population': m.get('population') or None,
        'area': m.get('area') or None,
        'scores': score_map,
        'social_media': meta_social
    })

with (HERE / 'city_data_export.json').open('w', encoding='utf-8') as f:
    json.dump({'cities': cities_out}, f, ensure_ascii=False, indent=2)

print('Wrote city_data_export.json with', len(cities_out), 'cities')
